i=0
with open('kraken2_log', 'w') as l:
    for f in forwardReads:
        r = f.replace("_R1_", "_R2_")
        name = "-".join(rSplit("/|-|_", f)[2:4])
     
        k2Cmd = f'kraken2 --db ~/SeqDBs/kraken/ --threads 16 \
        --report k2Reports/{name}.k2report --memory-mapping \
        --report-minimizer-data --minimum-hit-groups 3 --paired {f} {r} > k2Reports/{name}.kraken2'
        
        # rather than printing these outputs in this notebook redirect it to a  file
        kraken = Popen(k2Cmd, shell=True, stdout=PIPE, stderr=PIPE)
        i += 1
        stdout = kraken.communicate()
        stderr = kraken.communicate()
        l.write(f'{name}\n')
        l.write(f'{k2Cmd}\n')
        l.write(f'{stdout}\n')
        l.write(f'{stderr}\n')
        
        clear_output(wait=True)
        print(f'sample {i} out of {len(forwardReads)}')

Calculate diversities

From https://github.com/jenniferlu717/KrakenTools/tree/master/DiversityTools

This program calculates alpha diversity, from the Bracken abundance estimation file. User must specify Bracken output file, and type of alpha diversity to be calculated. Specific options are specified below.

\(\alpha\) diversities

!python /home/armand/software/KrakenTools/DiversityTools/alpha_diversity.py -f bReports/VRM-10A.bracken -a Sh
Shannon's diversity: 2.8115814392156477
def alpha(f, method):
    cmd = f'./alpha_diversity.py -f {f} -a {method}'
    alpha = Popen(cmd, shell=True, stdout=PIPE, stderr=PIPE)
    a = alpha.communicate()
    return a

brakenList = glob("bReports/*.bracken")
alphaDict = {}

for i in brakenList:
    a = alpha(i, 'Sh')[0].decode("utf-8").strip().split(": ")[-1]
    i = path.split(i)[-1]
    alphaDict[i] = float(a)

  
alphaDf = pd.DataFrame.from_dict(alphaDict, orient='index', columns=['alpha']).reset_index()
alphaDf.rename({'index': 'sample'}, axis=1, inplace=True)
alphaDf['type'] = alphaDf['sample'].apply(lambda s: s.split("-")[0])
alphaDf['pt'] = alphaDf['sample'].apply(lambda s: int(findall(r'\d+',s)[0]))
alphaDf['repl'] = alphaDf['sample'].apply(lambda s: rSplit("-|\.", s)[1][-1])
alphaDf.sort_values(by=['pt', 'repl'], inplace=True)
